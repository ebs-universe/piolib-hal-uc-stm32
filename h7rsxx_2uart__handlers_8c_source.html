23static inline void _uart_handler_tx_norm(const HAL_BASE_t intfnum){
24 // TXE in single byte mode
25 uint8_t wbyte;
26 if (bytebuf_cPopulation(uart_if[intfnum]->txbuf) <= 1){
27 *(HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_CR1) &= ~USART_CR1_TXEIE;
28 }
29 if (bytebuf_cPopulation(uart_if[intfnum]->txbuf)) {
30 wbyte = bytebuf_cPopByte(uart_if[intfnum]->txbuf);
31 *(HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_TDR) = wbyte;
32 }
33}

47static inline void _uart_handler_tx_fifo(const HAL_BASE_t intfnum){
48 // TXFT in fifo mode
49 // upto uC_UART2_HWF_TX_CSIZE bytes can be written
50 uint8_t wbyte;
51 uint8_t n = bytebuf_cPopulation(uart_if[intfnum]->txbuf);
52 while (n && (*(HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_ISR) & USART_ISR_TXE_TXFNF)){
53 wbyte = bytebuf_cPopByte(uart_if[intfnum]->txbuf);
54 *(HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_TDR) = wbyte;
55 n--;
56 }
57 if (!n){
58 *(HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_CR3) &= ~USART_CR3_TXFTIE;
59 }
60}

62static inline void _uart_handler_rx_fifo(const HAL_BASE_t intfnum){
63 // RXFT in fifo mode
64 // Ideally, upto uC_UART2_HWF_RX_CSIZE bytes can be read.
65 // This just does a byte by byte read, though, almost exactly like norm.
66 // In this case we read multiple bytes if they happen to exist in the
67 // FIFO, though in practice that should never actually happen.
68 //
69 // TODO RX FIFO Threshold Interrupt never seems to fire unless TX is also
70 // going on. Unsure why.
71 //
72 // TODO FIFO RX needs all messages to be a multiple of CSIZE, which can
73 // never be guaranteed. If < CSIZE bytes remain at message end,
74 // the interrupt will never fire. Use RTO or something to handle
75 // partial chunk reception, maybe? For the moment, we set CSIZE
76 // to 1, which makes this whole fifo thing pointless.
77 //
78 // TODO Overrun and various reception errors are not handled.
79 //
80 uint8_t rbyte;
81 uint8_t n = bytebuf_cGetFree(uart_if[intfnum]->rxbuf);
82 while (n && (*(HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_ISR) & USART_ISR_RXNE_RXFNE)) {
83 rbyte = (uint8_t) *(HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_RDR);
84 bytebuf_cPushByte(uart_if[intfnum]->rxbuf, rbyte, 0x00);
85 n--;
86 }
87}

89static inline void _uart_handler_norm(const HAL_BASE_t intfnum){
90 uint32_t flags = *(HAL_SFR_t *)(uart_if[intfnum]->hwif->base + OFS_UART_ISR);
91 if (flags & USART_ISR_TXE_TXFNF){
92 _uart_handler_tx_norm(intfnum);
93 }
94 if (flags & USART_ISR_RXNE_RXFNE){
95 _uart_handler_rx_norm(intfnum);
96 }
