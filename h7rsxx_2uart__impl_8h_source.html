22/*
23 * Implementation Notes
24 *
25 * Wordlength and Parity
26 * ---------------------
27 *
28 * This implementation presently requires the application to correctly
29 * set the wordlength, compensating for the parity bit, if parity is enabled.
30 * The defaults work fine for 8bit word, no parity. However, if parity is
31 * to be enabled, the application must configure the wordlength to 9 bit for
32 * what is actually 8 bit data.
33 *
34 * This may be improved at a later date.
35 *
36 * GPIO Configuration
37 * ------------------
38 *
39 * This implementation does not do anything to configure GPIO. Needed GPIO
40 * configuration should be done by the application, typically in bsp/gpio.
41 *
42 * Using the Hardware FIFO
43 * -----------------------
44 *
45 * Some basic implementation has been tried, but it does not work. Pure
46 * continuous tx without rx seems to work. rx through fifo threshold seems
47 * to work only while the tx is going on.
48 *
49 * The implementation tries to use the normal FIFO interrupts, and while
50 * rx itself works, tx does not seem to start under these circumstances.
51 *
52 * I'm giving up on this for the moment until there is a robust use case
53 * for the hardware fifo.
54 *
55 * Configuring for Speed
56 * ---------------------
57 *
58 * The default implementation prioritizes reliability. If speed is required
59 * instead, the following changes may be considered:
60 * - Enable the HW FIFO.
61 * - Reduce oversampling to 8
62 * - Consider use of the DMA. Doing so with the ebs bytebuf
63 * implementation will be quite difficult, though, so this may need
64 * to be reserved for interfaces which do not expect an ebs bytebuf.
65 *
66 * Bit Majority / One Bit
67 * ----------------------
68 *
69 * By default, we use majority vote to deal with potentially noisy
70 * environments. However, this reduces tolerance to clock deviations.
71 * This is a less clear tradeoff to make. This default is based on
72 * the assumption that bauds rate and crystals / clocks will be
73 * matched to generate accurate baud rates on both sides.
74 *
75 * At present, we don't actually do anything if the three measurements
76 * are not equal. The hardware simply provides us with the majority.
77 *
78 */

124static inline HAL_BASE_t uart_putc(HAL_BASE_t intfnum, uint8_t byte, uint8_t token, HAL_BASE_t handlelock){
125 uint8_t stat=1;
126 if (handlelock){
127 stat = bytebuf_cPushReqLock(uart_if[intfnum]->txbuf, 1, token);
128 }
129 if (stat){
130 stat = bytebuf_cPushByte(uart_if[intfnum]->txbuf, byte, token);
131 uart_send_trigger(intfnum);
132 return stat;
133 }
134 return 0;
135}

160static inline void uart_send_trigger(HAL_BASE_t intfnum){
161 if (uart_if[intfnum]->hwif->fifomode){
162 if (!(* (HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_CR3) & USART_CR3_TXFTIE)) {
163 * (HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_CR3) |= USART_CR3_TXFTIE;
164 }
165 } else {
166 if (!(* (HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_CR1) & USART_CR1_TXEIE)){
167 * (HAL_SFR_t *) (uart_if[intfnum]->hwif->base + OFS_UART_CR1) |= USART_CR1_TXEIE;
168 }
169 }
170}

172static inline void uart_send_flush(HAL_BASE_t intfnum){
