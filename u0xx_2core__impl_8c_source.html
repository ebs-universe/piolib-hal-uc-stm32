66 HAL_StatusTypeDef hal_rcc_clockconfig(const RCC_ClkInitTypeDef *const RCC_ClkInitStruct, uint32_t FLatency)
67 {
68  HAL_StatusTypeDef halstatus;
69  uint32_t tickstart;
70 
71  /* Check the parameters */
72  assert_param(RCC_ClkInitStruct != NULL);
73  assert_param(IS_RCC_CLOCKTYPE((uint8_t)RCC_ClkInitStruct->ClockType));
74  assert_param(IS_FLASH_LATENCY(FLatency));
75 
76  /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
77  must be correctly programmed according to the frequency of the CPU clock
78  (HCLK) and the supply voltage of the device. */
79 
80  /* Increasing the number of wait states because of higher CPU frequency */
81  if (FLatency > __HAL_FLASH_GET_LATENCY())
82  {
83  /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
84  __HAL_FLASH_SET_LATENCY(FLatency);
85 
86  /* Check that the new number of wait states is taken into account to access the Flash
87  memory by reading the FLASH_ACR register */
88  uint32_t timeout = 100;
89 
90  while (__HAL_FLASH_GET_LATENCY() != FLatency)
91  {
92  timeout --;
93  }
94 
95  if (!timeout) {
96  die();
97  }
98  }
99 
100  /*------------------------- SYSCLK Configuration ---------------------------*/
101  if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_SYSCLK) == RCC_CLOCKTYPE_SYSCLK)
102  {
103  assert_param(IS_RCC_SYSCLKSOURCE(RCC_ClkInitStruct->SYSCLKSource));
104 
105  /* PLL is selected as System Clock Source */
106  if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_PLLCLK)
107  {
108  /* Check the PLL ready flag */
109  if (READ_BIT(RCC->CR, RCC_CR_PLLRDY) == 0U)
110  {
111  die();
112  }
113  }
114  else
115  {
116  if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_HSE)
117  {
118  /* Check the HSE ready flag */
119  if (READ_BIT(RCC->CR, RCC_CR_HSERDY) == 0U)
120  {
121  die();
122  }
123  }
124  /* MSI is selected as System Clock Source */
125  else if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_MSI)
126  {
127  /* Check the MSI ready flag */
128  if (READ_BIT(RCC->CR, RCC_CR_MSIRDY) == 0U)
129  {
130  die();
131  }
132  }
133  /* HSI is selected as System Clock Source */
134  else if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_HSI)
135  {
136  /* Check the HSI ready flag */
137  if (READ_BIT(RCC->CR, RCC_CR_HSIRDY) == 0U)
138  {
139  die();
140  }
141  }
142 
143  /* LSI is selected as System Clock Source */
144  else if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_LSI)
145  {
146  /* Check the LSI ready flag */
147  if (READ_BIT(RCC->CSR, RCC_CSR_LSIRDY) == 0U)
148  {
149  die();
150  }
151  }
152 
153  /* LSE is selected as System Clock Source */
154  else
155  {
156  /* Check the LSE ready flag */
157  if (READ_BIT(RCC->BDCR, RCC_BDCR_LSERDY) == 0U)
158  {
159  die();
160  }
161  }
162  }
163 
164  MODIFY_REG(RCC->CFGR, RCC_CFGR_SW, RCC_ClkInitStruct->SYSCLKSource);
165 
166  /* Get Start Tick */
167  tickstart = HAL_GetTick();
168 
169  while (__HAL_RCC_GET_SYSCLK_SOURCE() != (RCC_ClkInitStruct->SYSCLKSource << RCC_CFGR_SWS_Pos))
170  {
171  if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
172  {
173  die();
174  }
175  }
176  }
177 
178  /*-------------------------- HCLK Configuration --------------------------*/
179  if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_HCLK) == RCC_CLOCKTYPE_HCLK)
180  {
181  /* Set the new HCLK clock divider */
182  assert_param(IS_RCC_HCLK(RCC_ClkInitStruct->AHBCLKDivider));
183  MODIFY_REG(RCC->CFGR, RCC_CFGR_HPRE, RCC_ClkInitStruct->AHBCLKDivider);
184  }
185 
186  /* Decreasing the number of wait states because of lower CPU frequency */
187  if (FLatency < __HAL_FLASH_GET_LATENCY())
188  {
189  /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
190  __HAL_FLASH_SET_LATENCY(FLatency);
191 
192  /* Check that the new number of wait states is taken into account to access the Flash
193  memory by reading the FLASH_ACR register */
194  if (__HAL_FLASH_GET_LATENCY() != FLatency)
195  {
196  die();
197  }
198  }
199 
200  /*-------------------------- PCLK Configuration ---------------------------*/
201  if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_PCLK1) == RCC_CLOCKTYPE_PCLK1)
202  {
203  assert_param(IS_RCC_PCLK(RCC_ClkInitStruct->APB1CLKDivider));
204  MODIFY_REG(RCC->CFGR, RCC_CFGR_PPRE, RCC_ClkInitStruct->APB1CLKDivider);
205  }
206 
207  /* Update the SystemCoreClock global variable */
208  SystemCoreClock = HAL_RCC_GetSysClockFreq() >> (AHBPrescTable[READ_BIT(RCC->CFGR, RCC_CFGR_HPRE) \
209  >> RCC_CFGR_HPRE_Pos] & 0x1FU);
210 
211  /* Configure the source of time base considering new system clocks settings*/
212  halstatus = HAL_InitTick(TICK_INT_PRIORITY);
213 
214  return halstatus;
215 }
