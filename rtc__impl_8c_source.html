38 static uint8_t BCD_to_binary(uint8_t bcd) {
39  return ((bcd >> 4) * 10) + (bcd & 0x0F);

42 static uint8_t binary_to_BCD(uint8_t bin) {
43  return ((bin / 10) << 4) | (bin % 10);

95 static inline void _rtc_read(){
96  if (!rtc_state.iobuffer){
97  return;
98  }
99  uint32_t tr = RTC->TR;
100  uint32_t dr = RTC->DR;
101 
102  rtc_state.iobuffer->century = 20;
103  rtc_state.iobuffer->year = BCD_to_binary((dr >> RTC_DR_YU_Pos) & 0xFF);
104  rtc_state.iobuffer->month = BCD_to_binary((dr >> RTC_DR_MU_Pos) & 0x1F);
105  rtc_state.iobuffer->date = BCD_to_binary((dr >> RTC_DR_DU_Pos) & 0x3F);
106  rtc_state.iobuffer->hours = BCD_to_binary((tr >> RTC_TR_HU_Pos) & 0x3F);
107  rtc_state.iobuffer->minutes = BCD_to_binary((tr >> RTC_TR_MNU_Pos) & 0x7F);
108  rtc_state.iobuffer->seconds = BCD_to_binary((tr >> RTC_TR_SU_Pos) & 0x7F);
109 
110  // SSR register can be used to handle some subsecond correction.
111  // This is not implemented here though.
112 }

114 static inline void _rtc_write() {
115  if (!rtc_state.iobuffer){
116  return;
117  }
118  uint32_t year_bcd = binary_to_BCD(rtc_state.iobuffer->year);
119  uint32_t month_bcd = binary_to_BCD(rtc_state.iobuffer->month);
120  uint32_t date_bcd = binary_to_BCD(rtc_state.iobuffer->date);
121  uint32_t hours_bcd = binary_to_BCD(rtc_state.iobuffer->hours);
122  uint32_t minutes_bcd = binary_to_BCD(rtc_state.iobuffer->minutes);
123  uint32_t seconds_bcd = binary_to_BCD(rtc_state.iobuffer->seconds);
124 
125  RTC->TR = (hours_bcd << RTC_TR_HU_Pos) |
126  (minutes_bcd << RTC_TR_MNU_Pos) |
127  (seconds_bcd << RTC_TR_SU_Pos);
128 
129  RTC->DR = (year_bcd << RTC_DR_YU_Pos) |
130  (month_bcd << RTC_DR_MU_Pos) |
131  (date_bcd << RTC_DR_DU_Pos);
132 }

134 static void _rtc_hw_init(){
135  // TAMP and ALARM code can simply unlock and configure as needed. As
136  // long as it does not push into the init mode, the RTC should remain
137  // unaffected.
138  _rtc_unlock();
139  RTC->ICSR |= RTC_ICSR_INIT;
140  while (!(RTC->ICSR & RTC_ICSR_INITF));
141 
142  RTC->PRER = 255 | (127 << RTC_PRER_PREDIV_A_Pos);
143 
144  #if uC_RTC_USE_MIXEDMODE
145  RTC->ICSR = (RTC->ICSR & ~RTC_ICSR_BIN_Msk) | RTC_ICSR_BIN_1;
146  // TODO BCDU needs to be calculated and set here
147  #else
148  RTC->ICSR = (RTC->ICSR & ~(RTC_ICSR_BIN_Msk | RTC_ICSR_BCDU_Msk));
149  #endif
150  RTC->CR &= ~RTC_CR_FMT_Msk;
151 
152  _rtc_write();
153 
154  RTC->ICSR &= ~RTC_ICSR_INIT;
155  _rtc_lock();
156 }

158 void rtc_init(void) {
159  __HAL_RCC_PWR_CLK_ENABLE();
160  HAL_PWR_EnableBkUpAccess();
161  while(!(PWR->CR1 & PWR_CR1_DBP));
162  // clk config probably isnt needed on every init
163  __HAL_RCC_RTC_CONFIG(RCC_RTCCLKSOURCE_LSE);
164  __HAL_RCC_RTC_ENABLE();
165  __HAL_RCC_RTCAPB_CLK_ENABLE();
166  if (!(RTC->ICSR & RTC_ICSR_INITS)){
167  // Year field is zero. Not Initialized.
168  _rtc_hw_init();
169  } else {
170  _rtc_wait_sync();
171  _rtc_read();
172  }
173  #if uC_RTC_1HZ_OUT_ENABLED
174  _rtc_1hz_out_init();
175  #endif
176  _rtc_state_reset();
177 }

179 uint8_t rtc_read(tm_real_t * target, void(*cb)(void)){
180  if (rtc_state.state != RTC_ST_IDLE) {
181  return 2;
182  }
183  memset(target, 0, sizeof(tm_real_t));
184  rtc_state.iobuffer = target;
185  rtc_state.state = RTC_ST_WAIT_READ;
186  _rtc_wait_sync();
187  _rtc_read();
188  if (cb) { cb(); }
189  _rtc_state_reset();
190  return 0;
191 }

196  }

198  rtc_state.state = RTC_ST_WAIT_WRITE;
199  _rtc_hw_init();
200  if (cb) { cb(); }
201  _rtc_state_reset();
202  return 0;
