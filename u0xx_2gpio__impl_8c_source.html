78void gpio_conf_periph(
79 PORTSELECTOR_t port, PINSELECTOR_t pin, uint8_t periph)
80{
81 HAL_SFR_t *moder = (HAL_SFR_t *)(port + GPIO_OFS_MODE);
82 *moder = (*moder & ~(3U << (pin * 2))) | (2U << (pin * 2));
83
84 HAL_SFR_t *afr = (HAL_SFR_t *)(port + ((pin < 8) ? GPIO_OFS_AF0 : GPIO_OFS_AF1));
85 uint32_t afr_position = (pin % 8) * 4;
86 *afr = (*afr & ~(0xF << afr_position)) | ((periph & 0xF) << afr_position);
87}

129void gpio_conf_interrupt(
130 PORTSELECTOR_t port, PINSELECTOR_t pin, HAL_BASE_t edge)
131{
132 // Configure EXTI Source Port Selection
133 HAL_SFR_t *exticr = (HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_EXTICR1 + 4U * (pin >> 2));
134 uint32_t bit_position = (pin & 0x03U) * 8U;
135 *exticr = (*exticr & ~(0xFFU << bit_position)) | (gpio_get_port_index(port) << bit_position);
136
137 // Configure Trigger Edge Selection
138 if (edge & GPIO_INT_EDGE_RISING) {
139 *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_RTSR1) |= (1 << pin);
140 } else {
141 *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_RTSR1) &= ~(1 << pin);
142 }
143
144 if (edge & GPIO_INT_EDGE_FALLING) {
145 *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_FTSR1) |= (1 << pin);
146 } else {
147 *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_FTSR1) &= ~(1 << pin);
148 }
149}
