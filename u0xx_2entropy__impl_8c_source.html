29 uint8_t entropy_buffer[uC_ENTROPY_POOL_SIZE];
30 bytebuf entropy_pool;
31 entropy_state_t entropy_state;

33 static inline void _entropy_state_init(void){
34  if (!entropy_pool._size) {
35  bytebuf_vInit(entropy_buffer, uC_ENTROPY_POOL_SIZE, &entropy_pool);
36  }
37  entropy_state.entropy = &entropy_pool;
38 }
39 
40 static inline void _entropy_clocks_init(void){
41  // TODO This may need to be modified to use clock configuration
42  // infrastructure in core instead of hardcoding STMCube HAL calls
43  RCC_PeriphCLKInitTypeDef clk_cfg_rng = {0};
44  clk_cfg_rng.PeriphClockSelection = RCC_PERIPHCLK_RNG;
45  clk_cfg_rng.RngClockSelection = RCC_RNGCLKSOURCE_MSI;
46  if (HAL_RCCEx_PeriphCLKConfig(&clk_cfg_rng) != HAL_OK){
47  die();
48  }
49 }

51 /*
52  * Configure for NIST SP 800â€‘90B per ST AN4230
53  * CR 0x00F00D00
54  * HTCR 0xAAC7 @ 48MHz RNG clk
55  * Note that the datasheet suggests this is is configuration C, optimized
56  * for speed. Validation may be needed for critical applications.
57  *
58  * Additionally, we set bit 13 of CR since we expect the TRNG will typically
59  * be left off post entropy pool filling. This may need to be configurable.
60  *
61  * This function is provided by the HAL implementation in addition to the
62  * usual API for use in case the hardware reports a seed error and the TRNG
63  * needs to be reset.
64  *
65  * We don't actually do the reset if the entropy buffer is full for modes 1
66  * and 2. Here, just deinit the TRNG until it is needed next.
67  */
68 void entropy_reset(void){
69  #if uC_ENTROPY_MODE <= 2
70  if (!bytebuf_cGetFree(entropy_state.entropy)){
71  return;
72  }
73  #endif
74  __HAL_RCC_RNG_CLK_ENABLE();
75  NVIC_DisableIRQ(RNG_CRYP_IRQn);
76  RNG->CR &= ~(RNG_CR_RNGEN | RNG_CR_IE);
77  RNG->CR |= (RNG_CR_CONDRST |
78  (0x0F << RNG_CR_RNG_CONFIG1_Pos) |
79  (0x0D << RNG_CR_RNG_CONFIG3_Pos) |
80  (0x01 << RNG_CR_RNG_CONFIG2_Pos));
81  RNG->HTCR = 0xAAC7;
82  RNG->CR &= ~RNG_CR_CONDRST;
83  RNG->CR |= (RNG_CR_RNGEN | RNG_CR_IE);
84  NVIC_EnableIRQ(RNG_CRYP_IRQn);
85  entropy_state.accumulating = 1;
86 }
87 
88 void entropy_init(void) {
89  _entropy_state_init();
90  _entropy_clocks_init();
91  HAL_NVIC_SetPriority(RNG_CRYP_IRQn, 0, 0);
92  entropy_reset();
93 }
94 
95 void entropy_deinit(void) {
96  RNG->CR &= ~(RNG_CR_RNGEN | RNG_CR_IE);
97  __HAL_RCC_RNG_CLK_DISABLE();
98  entropy_state.accumulating = 0;
99 }
