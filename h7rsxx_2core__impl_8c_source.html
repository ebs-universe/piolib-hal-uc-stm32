30__weak void mpu_init(void){
31 MPU_Region_InitTypeDef MPU_InitStruct = {0};
32
33 HAL_MPU_Disable();
34
35 // Background protection for preventing speculative execution
36 MPU_InitStruct.Enable = MPU_REGION_ENABLE;
37 MPU_InitStruct.Number = MPU_REGION_NUMBER0;
38 MPU_InitStruct.BaseAddress = 0x0;
39 MPU_InitStruct.Size = MPU_REGION_SIZE_4GB;
40 MPU_InitStruct.SubRegionDisable = 0x87;
41 MPU_InitStruct.TypeExtField = MPU_TEX_LEVEL0;
42 MPU_InitStruct.AccessPermission = MPU_REGION_NO_ACCESS;
43 MPU_InitStruct.DisableExec = MPU_INSTRUCTION_ACCESS_DISABLE;
44 MPU_InitStruct.IsShareable = MPU_ACCESS_SHAREABLE;
45 MPU_InitStruct.IsCacheable = MPU_ACCESS_NOT_CACHEABLE;
46 MPU_InitStruct.IsBufferable = MPU_ACCESS_NOT_BUFFERABLE;
47 HAL_MPU_ConfigRegion(&MPU_InitStruct);
48
49 MPU_InitStruct = (MPU_Region_InitTypeDef){0};
50
51 // Read Access for ID
52 MPU_InitStruct.Number = MPU_REGION_NUMBER1;
53 MPU_InitStruct.BaseAddress = 0x08fff000;
54 MPU_InitStruct.Size = MPU_REGION_SIZE_4KB;

67void prepare_tcm_memory(void){
68 /* Copy ITCM code */
69 extern const unsigned char _sitcm;
70 extern const unsigned char _eitcm;
71 extern const unsigned char _siitcm;
72 extern const unsigned char _sdtcm;
73 extern const unsigned char _edtcm;
74
75 void * sitcm = (void *)&_sitcm;
76 void * eitcm = (void *)&_eitcm;
77 void * siitcm = (void *)&_siitcm;
78 void * sdtcm = (void *)&_sdtcm;
79 void * edtcm = (void *)&_edtcm;
80
81 int size = eitcm - sitcm;
82 if (size) {
83 memcpy(sitcm, siitcm, size);
84 }
85 size = edtcm - sdtcm;
86 if (size) {
87 memset(sdtcm, 0, size);
88 }
89}

93__weak void configure_memory_layout(void){
94 FLASH_OBProgramInitTypeDef pOBInit = {0};
95
96 if (HAL_FLASH_Unlock() != HAL_OK) { die(); }
97 if (HAL_FLASH_OB_Unlock() != HAL_OK) { die(); }
98
99 pOBInit.OptionType = OPTIONBYTE_USER;
100 pOBInit.USERType = OB_USER_SRAM_ECC
101 | OB_USER_I2C_NI3C
102 | OB_USER_DTCM_AXI_SHARE
103 | OB_USER_ITCM_AXI_SHARE;
104
105 pOBInit.USERConfig2 = OB_I2C_NI3C_I2C
106 |OB_DTCM_AXI_SHARE_1
107 |OB_ITCM_AXI_SHARE_0
108 |OB_AXISRAM_ECC_DISABLE;
109
110 if (HAL_FLASHEx_OBProgram(&pOBInit) != HAL_OK) { die(); }
111 if (HAL_FLASH_OB_Lock() != HAL_OK) { die(); }
112 if (HAL_FLASH_Lock() != HAL_OK) { die(); }
113 prepare_tcm_memory();
114}
