268void dma_enqueue(HAL_BASE_t intfnum, dma_lli_t * lli){
269 if (!dma_chn_if[intfnum].state->ll_pending){
270 dma_chn_if[intfnum].state->ll_pending = lli;
271 return;
272 }
273
274 uint32_t * walker = &(dma_chn_if[intfnum].state->ll_pending->llr);
275 uint32_t lbar = *(HAL_SFR_t *)(dma_chn_if[intfnum].hwif.base + OFS_DMA_Cx_LBAR);
276
277 while (*walker) {
278 dma_lli_t * next = (dma_lli_t *)(lbar | (*walker & 0x0000FFFC)) ;
279 walker = &(next->llr);
280 }
281
282 // Attach the provided item to the end
283 const uint32_t llspec = (DMA_CLLR_UT1 | DMA_CLLR_UT2 | DMA_CLLR_UB1 | DMA_CLLR_USA | DMA_CLLR_UDA | DMA_CLLR_ULL);
284 *walker = llspec | (((uint32_t)lli) & 0x0000FFFC);
285}

288dma_lli_t* dma_setup_xfer(HAL_BASE_t intfnum, dma_xfer_t * xfer, dma_flow_control_t * fc, uint8_t token){
289 dma_lli_t * lli = _dma_request_lli(token);
290
291 // Transfer Configuration
292
293 lli->tr1 = 0;
294 lli->tr1 |= ((uint32_t)(xfer->src_opts & DMA_SRC_OPTS_MSK) >> DMA_SRC_OPTS_POS) << DMA_TR1_SRC_OPTS_POS;
295 lli->tr1 |= ((uint32_t)(xfer->dst_opts & DMA_DST_OPTS_MSK) >> DMA_DST_OPTS_POS) << DMA_TR1_DST_OPTS_POS;
296 lli->tr1 |= ((uint32_t)(xfer->dst_burst_len) << DMA_CTR1_DBL_1_Pos);
297 lli->tr1 |= ((uint32_t)(xfer->src_burst_len) << DMA_CTR1_SBL_1_Pos);
298 lli->tr1 |= ((uint32_t)(xfer->src_opts & DMA_OPT_WIDTH_MSK) >> DMA_OPT_WIDTH_POS) << DMA_CTR1_SDW_LOG2_Pos;
299 lli->tr1 |= ((uint32_t)(xfer->dst_opts & DMA_OPT_WIDTH_MSK) >> DMA_OPT_WIDTH_POS) << DMA_CTR1_DDW_LOG2_Pos;
300
305 switch(xfer->type){
306 case DMA_XFER_MEMORY_MEMORY:
307 lli->tr1 |= (DMA_CTR1_SINC | DMA_CTR1_DINC);
308 break;
309 case DMA_XFER_MEMORY_PERIPH:
310 lli->tr1 |= DMA_CTR1_SINC;
311 break;
312 case DMA_XFER_PERIPH_MEMORY:
313 lli->tr1 |= DMA_CTR1_DINC;
314 break;
315 case DMA_XFER_PERIPH_PERIPH:
316 break;
317 default:
318 die();
319 }
320
325 lli->br1 = (uint32_t)(xfer->size & 0xFFFF) | ((xfer->repeats & 0x3FF) << DMA_CBR1_BRC_Pos);
326 lli->sar = xfer->src;
327 lli->dar = xfer->dst;
328 lli->llr = 0;
329
330 // Flow Control Configuration
331
332 lli->tr2 = 0;
333 switch (fc->req_type){
334 case DMA_REQ_SOFTWARE:
335 lli->tr2 |= DMA_CTR2_SWREQ;
336 break;
337 case DMA_REQ_DST:
338 lli->tr2 |= DMA_CTR2_DREQ;
339 break;
340 case DMA_REQ_SRC:
341 case DMA_REQ_NONE:
342 break;
343 default:
344 die();
345 };
346
347 // We expect linear channels to use channel level event generation
348 // (end of LL). We do not support other event modes nor do we allow
349 // mixing of different transfer types in the same channel.
350 uint8_t event_mode = 0b11;
351
352 if (dma_chn_if[intfnum].hwif.type == DMA_CHN_TYPE_CIRCULAR){
353 // We expect circular channels to use LLI level event generation
354 // (end of each LLI). We do not support other event modes nor do
355 // we allow mixing of different transfer types in the same channel.
356 event_mode = 0b10;
357 }
358
359 uint8_t opts = fc->opts;
360
361 uint8_t event_mode_requested = (opts & DMA_FC_EVENT_MODE_MSK) >> DMA_FC_EVENT_MODE_POS;
362 if (event_mode_requested && event_mode_requested != event_mode) {
363 die();
364 }
365
366 if (opts & DMA_FC_OPT_REQ_BLOCK) {
367 lli->tr2 |= DMA_CTR2_BREQ;
368 }
369 lli->tr2 |= (uint32_t)(fc->req_sel);
370 lli->tr2 |= ((uint32_t)(opts & DMA_FC_TRIG_MODE_MSK) >> DMA_FC_TRIG_MODE_POS) << DMA_CTR2_TRIGM_Pos;
371 lli->tr2 |= ((uint32_t)(opts & DMA_FC_TRIG_POL_MSK) >> DMA_FC_TRIG_POL_POS) << DMA_CTR2_TRIGPOL_Pos;
372 lli->tr2 |= (uint32_t)(event_mode) << DMA_CTR2_TCEM_Pos;
373 lli->tr2 |= (uint32_t)(fc->trig_sel) << DMA_CTR2_TRIGSEL_Pos;
374
375 lli->cb_hc = fc->cb_hc;
376 lli->cb_tc = fc->cb_tc;
377
378 // 2D Addressing Configuration
379
380 // tr3, br2 have to do with 2d addressing and is not part of the linked list
381 // structure for the low numbered channels. we decide to simply not support it
382 // for now, and will need to deal with it separately when a suitable use case
383 // presents itself.
384
385 // HAL_SFR_t * br2 = (HAL_SFR_t *)(dma_chn_if[intfnum].hwif.base + OFS_DMA_Cx_BR2);
386 // *br2 = 0;
387
388 // HAL _SFR_t * tr3 = (HAL_SFR_t *)(dma_chn_if[intfnum].hwif.base + OFS_DMA_Cx_TR3);
389 // *tr3 = ((uint32_t)(xfer->dst_offset_incr) << 16) | (xfer->src_offset_incr)
390
391 dma_enqueue(intfnum, lli);
392 return lli;
393}
