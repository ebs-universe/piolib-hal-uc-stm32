16 static void _adc1_handler(void){
17  uint32_t flags = *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_ISR);
18  uint16_t data;
19 
20  if (flags & ADC_FLAG_OVR){
21  *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_ISR) |= ADC_ISR_OVR;
22  adc1_state.overrun ++;
23  }
24 
25  if (flags & ADC_FLAG_EOC){
26  data = *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_DR);
27  adc1_state.lastresult = data;
28 
29  // Dispatch result if handler is set
30  if (adc1_state.handler_eoc){
31  adc1_state.handler_eoc(adc1_state.nextchn, &adc1_state.lastresult);
32  }
33 
34  // Prepare for next
35  uint32_t seqstate = adc1_state.seqstate;
36  seqstate >>= 1;
37  if (seqstate){
38  uint8_t offset = __builtin_ctz(seqstate);
39  seqstate >>= offset;
40  adc1_state.nextchn += (1 + offset);
41  adc1_state.seqstate = seqstate;
42  }
43  }
44 
45  if (flags & ADC_FLAG_EOS){
46  // Clear EOS flag
47  *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_ISR) |= ADC_ISR_EOS;
48  if (adc1_state.mode != ADC_MODE_CONTINUOUS) {
49  adc1_state.mode = ADC_MODE_IDLE;
50  }
51 
52  if (adc1_state.handler_eos){
53  adc1_state.handler_eos();
54  }
55 
56  // Reset sequence state
57  adc1_state.nextchn = adc1_state.chnmask;
58  adc1_state.seqstate = adc1_state.firstchn;
59  }
60 }
