28static inline void _adc1_handler(void){
29 #ifdef uC_ADC_TIMING_MONITOR_GPIO
30 gpio_set_output_high(uC_ADC_TIMING_MONITOR_GPIO);
31 #endif
32
33 uint32_t flags = *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_ISR);
34
35 if (flags & ADC_FLAG_EOC){
36 adc1_state.lastresult = *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_DR);
37
38 // Dispatch result if handler is set
39 if (adc1_state.handler_eoc){
40 adc1_state.handler_eoc(adc1_state.nextchn, &adc1_state.lastresult);
41 }
42
43 // Prepare for next
44 uint32_t seqstate = adc1_state.seqstate;
45 seqstate >>= 1;
46 if (seqstate){
47 uint8_t offset = __builtin_ctz(seqstate);
48 seqstate >>= offset;
49 adc1_state.nextchn += (1 + offset);
50 adc1_state.seqstate = seqstate;
51 }
52 }
53
54 if (flags & ADC_FLAG_EOS){
55 // Clear EOS flag
56 *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_ISR) |= ADC_ISR_EOS;
57 if (adc1_state.mode != ADC_MODE_CONTINUOUS) {
58 adc1_state.mode = ADC_MODE_IDLE;
59 }
60
61 if (adc1_state.handler_eos){
62 adc1_state.handler_eos();
63 }
64
65 // Reset sequence state
66 adc1_state.nextchn = adc1_state.chnmask;
67 adc1_state.seqstate = adc1_state.firstchn;
68 }
69
70 if (flags & ADC_FLAG_OVR){
71 *(HAL_SFR_t *)(ADC1_BASE + OFS_ADCn_ISR) |= ADC_ISR_OVR;
72 adc1_state.overrun ++;
73 }

77 #endif
78}
