38static inline void _uart_conf_baud(
39 const _uart_hwif_t *const hwif,
40 uint32_t kerclk,
41 uint32_t baudrate,
42 uint32_t oversampling)
43{
44 // PRESC : not used
45 HAL_SFR_t * presc = (HAL_SFR_t *)(hwif->base + OFS_UART_PRESC);
46 *presc = 0x0000;
47 // BRR write the actual register
48 HAL_SFR_t * brr = (HAL_SFR_t *)(hwif->base + OFS_UART_BRR);
49 uint16_t usartdiv = kerclk / baudrate;
50 if (oversampling == UART_OVERSAMPLING_16){
51 *brr = usartdiv;
52 } else {
53 *brr = ((usartdiv & 0x000F) >> 1) | (usartdiv & 0xFFF0);
54 }
55}

57static inline uint32_t _exp_uart_fifomode(bool fifomode){
58 if (fifomode){
59 return UART_FIFOMODE_ENABLE;
60 } else {
61 return UART_FIFOMODE_DISABLE;
62 }
63}

65static inline void _uart_init(
66 const _uart_hwif_t *const hwif,
67 uint32_t kerclk,
68 uint32_t wordlength,
69 bool fifomode,
70 uint32_t txfifo_threshold,
71 uint32_t rxfifo_threshold,
72 uint32_t stopbits,
73 uint32_t baudrate,
74 uint32_t oversampling,
75 uint32_t onebit,
76 uint32_t parity,
77 uint32_t endianness,
78 uint32_t polarity_data,
79 uint32_t polarity_tx,
80 uint32_t polarity_rx,
81 uint32_t swap
82 )
83{
84 _uart_conf_baud(hwif, kerclk, baudrate, oversampling);
85
86 HAL_SFR_t * cr1 = (HAL_SFR_t *)(hwif->base + OFS_UART_CR1);
87 HAL_SFR_t * cr2 = (HAL_SFR_t *)(hwif->base + OFS_UART_CR2);
88 HAL_SFR_t * cr3 = (HAL_SFR_t *)(hwif->base + OFS_UART_CR2);
89 HAL_SFR_t * gtpr = (HAL_SFR_t *)(hwif->base + OFS_UART_GTPR);
90 HAL_SFR_t * rtor = (HAL_SFR_t *)(hwif->base + OFS_UART_RTOR);
91 HAL_SFR_t * rqr = (HAL_SFR_t *)(hwif->base + OFS_UART_RQR);
92
93
94 *cr1 = 0x00000000;
95 *cr2 = 0x00000000;
96 *cr3 = 0x00000000;
97
98 if (hwif->type == UART_HWIF_FULL){
99 *gtpr = 0x00000000; // unused
100 *rtor = 0x00000000; // unused
101 }
102
103 *rqr = 0x00000000;
104
105 /* CR1
106 * Provided:
107 * wordlength | fifomode | oversampling | parity
108 * (ue=0) (ue=0) (ue=0) (ue=0)
109 *
110 * Impl Hardcoded Defaults:
111 * deat | dedt | mme | wake | uesm
112 * all set to disable / 0
113 *
114 * Deferred:
115 * te | re | ue | all interrupt enables
116 */
117 *cr1 = wordlength | oversampling | parity |
118 _exp_uart_fifomode(fifomode) |
119 (0x00000000 & USART_CR1_DEAT_Msk) |
120 (0x00000000 & USART_CR1_DEDT_Msk) |
121 (0x00000000 & USART_CR1_MME_Msk) |
122 (0x00000000 & USART_CR1_WAKE_Msk) |
123 (0x00000000 & USART_CR1_UESM_Msk);
124
125 /* CR2
126 *
127 * (all (ue=0) unless otherwise specified)
128 *
129 * Provided :
130 * stopbits | endianness |
131 * polarity_data | polarity_tx | polarity_rx | swap
132 *
133 * Impl Hardcoded Defaults:
134 * add | rtoen | abrmode | abren | linen |
135 * clken | cpol | cpha | lbcl | lbdl | addm7 |
136 * dis_nss | slven
137 *
138 * all set to disable / 0
139 */
140 *cr2 = stopbits | endianness | polarity_data | polarity_tx | polarity_rx | swap |
141 (0x00000000 & USART_CR2_ADD_Msk) |
142 (0x00000000 & USART_CR2_RTOEN_Msk) |
143 (0x00000000 & USART_CR2_ABRMODE_Msk) |
144 (0x00000000 & USART_CR2_ABREN_Msk) |
145 (0x00000000 & USART_CR2_LINEN_Msk) |
146 (0x00000000 & USART_CR2_CLKEN_Msk) |
147 (0x00000000 & USART_CR2_CPOL_Msk) |
148 (0x00000000 & USART_CR2_CPHA_Msk) |
149 (0x00000000 & USART_CR2_LBCL_Msk) |
150 (0x00000000 & USART_CR2_LBDL_Msk) |
151 (0x00000000 & USART_CR2_ADDM7_Msk) |
152 (0x00000000 & USART_CR2_DIS_NSS_Msk) |
153 (0x00000000 & USART_CR2_SLVEN_Msk);
154
155 /* CR3
156 *
157 * (all (ue=0) unless otherwise specified)
158 *
159 * Provided:
160 * onebit | txfifo_threshold | rxfifo_threshold
161 * (fifomode) (fifomode)
162 *
163 * Impl Hardcoded Defaults:
164 * wus | scarcnt | dep | dem | ddre | ovrdis | ctse | rtse |
165 * dmat | dmar | scen | nack | hdsel | irlp | iren
166 * all set to disable / 0
167 *
168 */
169 *cr3 = onebit |
170 (0x00000000 & USART_CR3_WUS_Msk) |
171 (0x00000000 & USART_CR3_SCARCNT_Msk) |
172 (0x00000000 & USART_CR3_DEP_Msk) |
173 (0x00000000 & USART_CR3_DEM_Msk) |
174 (0x00000000 & USART_CR3_DDRE_Msk) |
175 (0x00000000 & USART_CR3_OVRDIS_Msk) |
176 (0x00000000 & USART_CR3_CTSE_Msk) |
177 (0x00000000 & USART_CR3_RTSE_Msk) |
178 (0x00000000 & USART_CR3_DMAT_Msk) |
179 (0x00000000 & USART_CR3_DMAR_Msk) |
180 (0x00000000 & USART_CR3_SCEN_Msk) |
181 (0x00000000 & USART_CR3_NACK_Msk) |
182 (0x00000000 & USART_CR3_HDSEL_Msk) |
183 (0x00000000 & USART_CR3_IRLP_Msk) |
184 (0x00000000 & USART_CR3_IREN_Msk);
185
186 if (fifomode){
187 *cr3 |= txfifo_threshold | rxfifo_threshold;
188 }
189
190 // Final Enable
191 *cr1 |= USART_CR1_UE_Msk | USART_CR1_TE | USART_CR1_RE;
192
193 /*
194 * Interrupt Enables
195 *
196 * Enable | Status | Availability | Description
197 * -------|--------|--------------|-----------------------------------
198 * RXFFIE | Unused | FULL+LP fifo | RX FIFO Completely Full
199 * TXFEIE | Unused | FULL+LP fifo | TX FIFO Completely Empty
200 * EOBIE | Unused | FULL | Smartcard Mode
201 * RTOIE | Unused | FULL | Reciever Timeout
202 * CMIE | Unused | ALL (?) | RX Single Character Match
203 * PEIE | Unused | ALL | RX Parity Error
204 * TXFNFIE| Unused | FULL+LP fifo | TX FIFO has atleast one space
205 * TXEIE | nrml | FULL+LP nrml | TX SR empty
206 * RXFNEIE| Unused | FULL+LP fifo | RX FIFO has atleast one byte
207 * RXNEIE | nrml | FULL+LP nrml | RX SR has data
208 * TCIE | fifo | ALL | TX Complete. TX FIFO and SR empty
209 * IDLEIE | Unused | ALL | Idle Line Detected
210 * LBDIE | Unused | FULL | LIN mode
211 * RXFTIE | fifo | FULL fifo | RX FIFO Threshold
212 * TCBGTIE| Unused | FULL | Smartcard Mode
213 * TXFTIE | fifo | FULL fifo | TX FIFO Threshold
214 * WUFIE | Unused | FULL+LP | Wakeup
215 * CTSIE | Unused | ALL | Hardware Flow Control CTS
216 * EIE | Unused | ALL | Various Errors
217 *
218 */
219 if (fifomode) {
220 // Use FIFO threshold interrupts when using the HW FIFO
221 *cr3 |= USART_CR3_TXFTIE | USART_CR3_RXFTIE;
222 *cr1 |= USART_CR1_RXNEIE;
223 // This flag could signal when the TX is actually done.
224 //*cr1 |= USART_CR1_TCIE;
225 } else {
226 // Use regular interrupts when not using the HW FIFO
227 *cr1 |= USART_CR1_TXEIE | USART_CR1_RXNEIE;
228 }
229}

237HAL_BASE_t uart_vprintf(HAL_BASE_t intfnum, const char *format, va_list args) {
238 uint8_t stat, lstat;
239 stat = bytebuf_cPushReqBlindLock(uart_if[intfnum]->txbuf, BYTEBUF_TOKEN_PRINT);
240 if (stat){
241 stat = vbprintf(uart_if[intfnum]->txbuf, format, args);
242 lstat = bytebuf_cPushRelinquishLock(uart_if[intfnum]->txbuf, BYTEBUF_TOKEN_PRINT);
243 uart_send_trigger(intfnum);
244 if (lstat){
245 return stat;
246 }
247 else{
248 return 0;
249 }
250 }
251 else{
252 va_end(args);
253 return 0;
254 }
255}
