90 uint8_t crypto_set_ctx(crypto_ctx_t * ctx, CRYPTO_CTX_MODE_t mode,
91  uint8_t (*ingest_cb)(crypto_ctx_t *const, uint8_t)){
92  uint8_t wait;
93  if (mode == CRYPTO_CTX_MODE_IDLE || mode == CRYPTO_CTX_MODE_SUSPENDED){
94  die();
95  }
96  if (crypto_state.state == CRYPTO_PREINIT){
97  crypto_init();
98  }
99  if (crypto_state.ctx != ctx || crypto_state.ctx->mode != mode){
100  // There is a change in context.
101  if (crypto_state.state == CRYPTO_BUSY){
102  // We're in an active transaction. Do nothing.
103  return 0;
104  }
105  if (crypto_state.ctx){
106  // Note that the old context might be the same as the new
107  // context, except the mode changed.
108  if (crypto_state.ctx != ctx) {
109  if (crypto_state.ctx->mode > CRYPTO_CTX_MODE_IDLE){
110  crypto_state.ctx->mode = CRYPTO_CTX_MODE_SUSPENDED;
111  // TODO Deal with the old context.
112  // Suspend.
113  }
114  }
115  }
116  crypto_state.ctx = ctx;
117  crypto_state.ctx->mode = mode;
118  crypto_state.ctx->ingest_cb = ingest_cb;
119  wait = crypto_apply_profile(ctx->profile, mode, ctx->phase);
120  if (!wait){
121  crypto_state.state = CRYPTO_READY;
122  return 2;
123  } else {
124  crypto_state.state = CRYPTO_BUSY;
125  return 1;
126  }
127  } else {
128  // Nothing to do.
129  return 3;
130  }
131 }

206 /*
207  * This should also execute or trigger any special phase-dependent
208  * actions for the profile, such as key derivation. If the action
209  * needed takes time, this function should trigger it, set a
210  * phase other than the commanded phase, and do something in the
211  * interrupt handler to get the actual phase to be set eventually.
212  * When that happens, this function should return the actual phase
213  * set. Applications should then poll the actual phase until it
214  * reaches the necessary phase.
215  *
216  * This function can expect that the crypto core's activated context
217  * is valid, other than the phase.
218  */
219 CRYPTO_CTX_PHASE_t _set_phase(CRYPTO_CTX_PHASE_t phase){
220  switch (crypto_state.ctx->profile->alg){
221  case CRYPTO_AES_CBC:
222  case CRYPTO_AES_ECB:
223  if (phase == CRYPTO_CTX_PHASE_INIT){
224  // TODO actually fire off the derivation here?
225  crypto_state.ctx->phase = CRYPTO_CTX_PHASE_DERIVATION;
226  }
227  break;
228  case CRYPTO_AES_CTR:
229  break;
230  case CRYPTO_AES_GCM:
231  case CRYPTO_AES_GMAC:
232  case CRYPTO_AES_CCM:
233  switch (phase){
234  case CRYPTO_CTX_PHASE_INIT:
235  AES->CR = (AES->CR & ~(AES_CR_GCMPH));
236  crypto_state.ctx->phase = CRYPTO_CTX_PHASE_PREPKEY;
237  break;
238  case CRYPTO_CTX_PHASE_HEADER:
239  AES->CR = (AES->CR & ~(AES_CR_GCMPH)) | (AES_CR_GCMPH_0);
240  break;
241  case CRYPTO_CTX_PHASE_PAYLOAD:
242  AES->CR = (AES->CR & ~(AES_CR_GCMPH)) | (AES_CR_GCMPH_1);
243  break;
244  case CRYPTO_CTX_PHASE_FINAL:
245  AES->CR |= (AES_CR_GCMPH_1 | AES_CR_GCMPH_1);
246  break;
247  default:
248  die();
249  }
250  break;
251  default:
252  die();
253  }
254  return crypto_state.ctx->phase;
255 }

257 static void _set_mode(
258  CRYPTO_CTX_MODE_t mode,
259  CRYPTO_ALG_t alg,
260  CRYPTO_CTX_PHASE_t phase)
261 {
262  uint32_t mode_bits= 0UL;
263 
264  switch (mode) {
265  case CRYPTO_CTX_MODE_ENCRYPTION:
266  mode_bits = 0x0UL;
267  break;
268  case CRYPTO_CTX_MODE_DECRYPTION:
269  switch (alg) {
270  case CRYPTO_AES_CBC:
271  case CRYPTO_AES_ECB:
272  if (phase == CRYPTO_CTX_PHASE_DERIVATION){
273  mode_bits = AES_CR_MODE_1 | AES_CR_MODE_0;
274  break;}
275  case CRYPTO_AES_CTR:
276  case CRYPTO_AES_GCM:
277  case CRYPTO_AES_GMAC:
278  case CRYPTO_AES_CCM:
279  mode_bits = AES_CR_MODE_1;
280  break;
281  default:
282  die();
283  }
284  break;
285  default:
286  die();
287  }
288  AES->CR = (AES->CR & ~(AES_CR_MODE)) | mode_bits;
289 }

315 uint8_t crypto_apply_profile(
316  const crypto_profile_t *const profile,
317  CRYPTO_CTX_MODE_t mode,
318  CRYPTO_CTX_PHASE_t phase
319 ){
320  AES->CR &= ~AES_CR_EN;
321  // Clear any error flags
322  AES->CR |= (0x1UL << 8) | (0x1UL << 7);
323  _set_swapping(profile->swap);
324  _set_keysize(profile->key_size);
325  _set_alg(profile->alg);
326  _load_key_and_iv(profile->alg);
327  CRYPTO_CTX_PHASE_t res = _set_phase(phase);
328  _set_mode(mode, profile->alg, res);
329 
330  AES->CR |= AES_CR_CCFIE;
331  AES->CR |= AES_CR_EN;
332  if (res != phase){
333  return 1;
334  } else {
335  return 0;
336  }
337 }

376 uint8_t _crypto_tr_ingest_cb(crypto_ctx_t *const ctx, uint8_t ingest_len){
377  uint8_t read_for_last = 0;
378  crypto_state_t const* s = &crypto_state;
379  crypto_profile_t const*const p = s->ctx->profile;
380  crypto_transaction_t *const t = s->transaction;
381 
382  t->state = CRYPTO_TR_INPROGRESS;
383 
384  if (s->ctx->phase == CRYPTO_CTX_PHASE_HEADER){
385  if (p->header_size > t->input_len_done){
386  uint16_t header_remain = p->header_size - t->input_len_done;
387  if (header_remain > 16) {
388  _crypto_tr_ingest(16);
389  } else {
390  _crypto_tr_ingest(header_remain);
391  }
392  } else if (p->header_size == t->input_len_done) {
393  // header is now done, switch to payload
394  crypto_set_ctx_phase(s->ctx, CRYPTO_CTX_PHASE_PAYLOAD);
395  // dout is also not valid right now
396  read_for_last = 1;
397  }
398  }
399 
400  if (s->ctx->phase == CRYPTO_CTX_PHASE_PAYLOAD){
401  if (!read_for_last && t->input_len_done > 0){
402  _crypto_tr_egest(ingest_len);
403  }
404 
405  uint16_t payload_remain = t->input_len - t->input_len_done;
406  if (payload_remain > 16) {
407  _crypto_tr_ingest(16);
408  } else if (payload_remain) {
409  _crypto_tr_ingest(payload_remain);
410  } else {
411  crypto_set_ctx_phase(s->ctx, CRYPTO_CTX_PHASE_FINAL);
412  read_for_last = 1;
413  }
414  }
415 
416  if (s->ctx->phase == CRYPTO_CTX_PHASE_FINAL){
417  if (read_for_last){
418  // The ingest length is incorrect here for the last block
419  AES->DINR = 0x0UL;
420  AES->DINR = p->header_size * 8;
421  AES->DINR = 0x0UL;
422  AES->DINR = (t->input_len - p->header_size) * 8;
423  } else {
424  //Whatever is read here is the mac.
425  _crypto_tr_egest(16);
426  crypto_reset_ctx(s->ctx);
427  t->state = CRYPTO_TR_DONE;
428  }
429  }
430 
431  return 0;
432 }
