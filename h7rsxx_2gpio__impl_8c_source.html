127 void gpio_conf_periph(PORTSELECTOR_t port, PINSELECTOR_t pin, uint8_t periph)
128 {
129  HAL_SFR_t *moder = (HAL_SFR_t *)(port + GPIO_OFS_MODE);
130  *moder = (*moder & ~(3U << (pin * 2))) | (2U << (pin * 2));
131 
132  HAL_SFR_t *afr = (HAL_SFR_t *)(port + ((pin < 8) ? GPIO_OFS_AF0 : GPIO_OFS_AF1));
133  uint32_t afr_position = (pin % 8) * 4;
134  *afr = (*afr & ~(0xF << afr_position)) | ((periph & 0xF) << afr_position);
135 }

176 void gpio_conf_interrupt(PORTSELECTOR_t port, PINSELECTOR_t pin, HAL_BASE_t edge)
177 {
178  // Configure EXTI Source Port Selection
179  uint8_t word_index = pin >> 2U;
180  HAL_SFR_t *exticr = (HAL_SFR_t *)(&(SBS->EXTICR[word_index]));
181  uint32_t bit_position = (pin & 0x03U) * 4U;
182  uint8_t port_index = gpio_get_port_index(port);
183  uint32_t writing = (*exticr & ~(0x0FU << bit_position)) | (port_index << bit_position);
184  *exticr = writing;
185 
186  uint32_t readback = *exticr;
187  (void) readback;
188 
189  // Configure Trigger Edge Selection
190  if (edge & GPIO_INT_EDGE_RISING) {
191  *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_RTSR1) |= (1 << pin);
192  } else {
193  *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_RTSR1) &= ~(1 << pin);
194  }
195 
196  if (edge & GPIO_INT_EDGE_FALLING) {
197  *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_FTSR1) |= (1 << pin);
198  } else {
199  *(HAL_SFR_t *)(EXTI_BASE + EXTI_OFS_FTSR1) &= ~(1 << pin);
200  }
201 }
